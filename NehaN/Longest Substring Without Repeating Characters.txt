class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        ans = []
        final = []
        idx = 0
        start_idx = 0
        while(idx < len(s)):
            if len(final) < len(ans):
                final = ans
            if len(ans):
                if s[idx] in ans:
                    idx = start_idx+1
                    ans = []
                else:
                    ans.append(s[idx])
                    idx += 1
            else:
                start_idx = idx
                ans.append(s[idx])
                idx += 1
        if len(final) < len(ans):
                final = ans
        return len(final)
